class Solution:

    def zigzagTraversal(self, grid: List[List[int]]) -> List[int]:

        traversal = []

        for i in range(len(grid)):

            row = grid[i]

            if i % 2 == 0:

                traversal.extend(row)

            else:

                traversal.extend(reversed(row))

        # Now select every other element starting from 0

        result = [traversal[j] for j in range(0, len(traversal), 2)]

        return result
